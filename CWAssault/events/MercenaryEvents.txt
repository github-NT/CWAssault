# White Mercenaries
country_event = {
	id = merc.1
	title = merc.1.t
	desc = merc.1.d
	picture = GFX_White_Mercenaries

	is_triggered_only = no
	trigger = {
	any_owned_state = { is_on_continent = africa }
		OR = {
	has_war= yes
	has_civil_war = yes
		}
	}
	fire_only_once = no
	
	mean_time_to_happen = {
		days = 140
	}

	option = {
		name = merc.1.a
		add_manpower= 3000
		}
	option = {
		name = merc.1.b

		}
}
