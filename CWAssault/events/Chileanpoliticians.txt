add_namespace = chilepoli

# Allende Won
country_event = {
	id = chilepoli.1
	title = chilepoli.1.t
	desc = chilepoli.1.d
	picture = GFX_La_Moneda
	
	major = yes
	fire_only_once = yes
	
	is_triggered_only = yes
	trigger = {
	tag = CHL
	}

	option = {
	name = chilepoli.1.a
	create_country_leader = {
		name = "Salvador Allende"
		desc = "POLITICS_PEDRO_AGUIRRE_CERDA_DESC"
		picture = "gfx/leaders/cold_majors/Allende.dds"
		expire = "1965.1.1"
		ideology = Socialist
		traits = {
			#
		}
	}
		add_popularity = { ideology = Socialism popularity = 0.1 }
		set_politics = { ruling_party = Socialism elections_allowed = yes }
		subtract_from_variable  = { var = CHL_army_loyalty value = 60}
		subtract_from_variable  = { var = CHL_general_loyalty value = 60}
	}
}
