	is_triggered_only = yes

	trigger = {
		tag = SYR
		has_country_leader = { ruling_only = yes name = "Hafez al-Assad" }
		date <1976.1.1
		has_completed_focus= SYR_interventionistpolicies
	}

	option = {
		name = syria.5.a
			random_list = {
                8 = { country_event = { id = syria.7 days = 1 }
                    modifier = {
                        add = 100
                        is_historical_focus_on = yes
                    }
                }
                2 = { country_event = { id = syria.6 days = 1 } 
                    modifier = {
                        factor = 0
                        is_historical_focus_on = yes
                    }
                }
            }
	}
	option = {
		name = syria.5.b
		add_political_power = -50
	}
}
